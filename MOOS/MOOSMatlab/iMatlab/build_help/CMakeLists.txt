cmake_minimum_required(VERSION 2.8)
find_package(MOOS 10.0)
FOREACH (dir ${MOOS_INCLUDE_DIRS})
    MESSAGE("INCLUDE:"${dir})
ENDFOREACH(dir)

get_property(MOOS_CONFIG TARGET MOOS PROPERTY IMPORTED_CONFIGURATIONS)
GET_TARGET_PROPERTY(MOOS_LIB_LOCATION MOOS IMPORTED_LOCATION_${MOOS_CONFIG})

MESSAGE("LIB:"${MOOS_LIB_LOCATION})
