   GenericSensor_MOOSApp.cpp
   GenericSensor_Info.cpp
   main.cpp
)

  ${MOOS_LIBRARIES}
  mbutil
  geometry 
  contacts
  apputil
  ${SYSTEM_LIBS}
)
